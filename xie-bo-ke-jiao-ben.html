写博客脚本

Thu 29 January 2015

用pelican,github,markdown写博客，好处:免费。
不便： make html, 实时备份content,output * 蛋疼的环境迁移问题

于是，我写了此脚本解决上述问题

go.py

import sh
import os
print("Starting...")
print("switched to following dir ")
os.chdir("/home/tulpar/workplace/blog/content/")
print(os.system("pwd"))
print("pulling content from remote origin master....")
os.system("git pull origin master")
print("pulling output from remote origin master....")
os.system("git pull origin master")
os.system("cp /home/tulpar/workplace/blog/work_dir/*.md /home/tulpar/workplace/blog/content/")
print("copied *.md from work_dir to content")
print("pushing content to remote origin master....")
os.system("git add .")
os.system("git commit -m 'By script'")
os.system("git push origin master")
print("switched to following dir ")
os.chdir("/home/tulpar/docker/")
print(os.system("pwd"))
os.system('docker run -it -v /home/tulpar/workplace/blog/content/:/home/tulpar/blog_new/blog/content -v /home/tulpar/workplace/blog/output/:/home/tulpar/blog_new/blog/output/ tulpar008/blog:latest /bin/bash -c "cd /home/tulpar/blog_new/blog/;make html"')
print("switched to following dir ")
os.chdir("/home/tulpar/workplace/blog/output/")
print(os.system("pwd"))
print("pushing output to remote origin master....")
os.system("git add .")
os.system("git commit -m 'By script'")
os.system("git push origin master")

.bashrc

alias go='python /home/tulpar/workplace/blog/work_dir/go.py'
